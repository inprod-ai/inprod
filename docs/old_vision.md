from dataclasses import dataclass
from enum import Enum
from typing import Optional
import asyncio

class Confidence(Enum):
    PROVEN = 100      # Type system / compiler verified
    VERIFIED = 95     # Pattern + external validation
    HIGH = 80         # Static analysis rules
    LIKELY = 70       # LLM semantic analysis

class ShipDecision(Enum):
    SHIP = "ship"                    # All required gates pass
    SHIP_WITH_WAIVER = "waiver"      # Some gates fail, owner can accept

class FixType(Enum):

class AICodeValidator:
    """The main analysis engine."""
    
    def __init__(self, config: ValidatorConfig):
        self.config = config
        self.layers = [
            FastScanLayer(),
            StructuralAnalysisLayer(),
            SemanticAnalysisLayer(
                cheap_model=config.haiku_client,
                deep_model=config.opus_client,
            ),
            SynthesisLayer(),
        ]
    
    async def analyze(self, codebase_path: str) -> ShipReport:
        """Run analysis pipeline."""
        
        context = AnalysisContext(path=codebase_path)
        
        # Layer 1: Fast scan (seconds)
        # Secrets, CVEs, package hallucinations
        await self.layers[0].analyze(context)
        
        # Layer 2: Structural analysis (seconds to minutes)
        # AST, types, patterns
        await self.layers[1].analyze(context)
        
        # Layer 3: Semantic analysis (minutes)
        # Tiered: cheap model first, escalate suspicious
        if context.needs_semantic_analysis:
            await self.layers[2].analyze(context)
        
        # Layer 4: Synthesis (seconds)
        # Generate ship decision
        report = await self.layers[3].synthesize(context)
        
        return report
    
    async def fix(
        self, 
        codebase_path: str, 
        mechanical_only: bool = True
    ) -> FixResult:
        """Apply fixes with safety contract."""
        
        report = await self.analyze(codebase_path)
        
        fixes_to_apply = []
        for finding in report.all_findings:
            if finding.fix:
                if mechanical_only and finding.fix.type != FixType.MECHANICAL:
                    continue
                if await self.validate_fix(finding.fix):
                    fixes_to_apply.append(finding.fix)
        
        return await self.apply_fixes(fixes_to_apply)
    
    async def validate_fix(self, fix: Fix) -> bool:
        """Validate fix before applying."""
        # 1. Apply to temp branch
        # 2. Check compilation
        # 3. Run existing tests
        # 4. Check for new issues
        return fix.validated


class AICodeRiskDetector:
    """Detector for AI-generated code issues."""
    
    def __init__(self, registry_client, llm_client):
        self.registry = registry_client
        self.llm = llm_client
    
    async def detect_hallucinations(
        self, 
        context: AnalysisContext
    ) -> list[Finding]:
        """Find non-existent packages."""
        findings = []
        
        for import_stmt in context.imports:
            exists = await self.registry.package_exists(import_stmt.package)
            
            if not exists:
                # Also check for typosquatting
                similar = await self.registry.find_similar(import_stmt.package)
                
                findings.append(Finding(
                    id=f"hallucination-{import_stmt.hash}",
                    category="ai_risk",
                    title="Non-existent package import",
                    description=f"Package '{import_stmt.package}' not found",
                    file_path=import_stmt.file,
                    line_start=import_stmt.line,
                    line_end=import_stmt.line,
                    confidence=Confidence.VERIFIED,
                    detection_method="package_registry_lookup",
                    fix=self.suggest_real_package(import_stmt, similar),
                    effort_range=(5, 15),
                ))
        
        return findings
    
    async def detect_template_slop(
        self, 
        context: AnalysisContext
    ) -> list[Finding]:
        """Find tutorial-style code in production."""
        patterns = [
            (r'password.*=.*["\']password', "Example password"),
            (r'TODO:?\s*(implement|fix|add)', "Unimplemented TODO"),
            (r'localhost:\d+', "Hardcoded localhost"),
        ]
        
        findings = []
        for pattern, description in patterns:
            for match in context.search(pattern):
                if not self.is_test_file(match.file):
                    findings.append(Finding(
                        id=f"template-{match.hash}",
                        category="ai_risk",
                        title=f"Template slop: {description}",
                        description="Tutorial/example code in production",
                        file_path=match.file,
                        line_start=match.line,
                        line_end=match.line,
                        confidence=Confidence.HIGH,
                        detection_method="pattern_matching",
                        fix=None,  # Requires context
                        effort_range=(10, 30),
                    ))
        
        return findings


class TieredLLMAnalysis:
    """Cost-controlled LLM analysis."""
    
    def __init__(self, cheap_model, deep_model):
        self.cheap = cheap_model   # Haiku/Flash
        self.deep = deep_model     # Opus/GPT-4
    
    async def analyze(self, file: SourceFile) -> AnalysisResult:
        # First pass: cheap model
        quick = await self.cheap.analyze(file)
        
        # 80% of files stop here
        if quick.suspicion_score < 0.3:
            return quick
        
        # Escalate suspicious code
        return await self.deep.analyze(file)


class LLMGuardrails:
    """Safety measures for LLM analysis."""
    
    async def prepare_context(self, file: SourceFile) -> str:
        content = file.content
        
        # 1. Redact secrets before sending
        content = self.redact_secrets(content)
        
        # 2. Check for prompt injection
        if self.detect_prompt_injection(content):
            raise SecurityError("Potential prompt injection detected")
        
        return content
    
    def redact_secrets(self, content: str) -> str:
        """Remove secrets before LLM analysis."""
        # Replace API keys, passwords, tokens
        return SECRETS_REGEX.sub("[REDACTED]", content)
    
    def detect_prompt_injection(self, content: str) -> bool:
        """Check for prompt injection attempts."""
        injection_patterns = [
            r"ignore.*previous.*instructions",
            r"disregard.*above",
            r"new.*instructions",
        ]
        return any(re.search(p, content, re.I) for p in injection_patterns)
